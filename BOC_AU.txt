--Bank Of China in Australia
--Bank of China Limited Sydney Branch was established in 1942 and was the first Chinese bank to establish operations in Australia.
--It has branches all over Australia including 5 branches in Sydney, 3 branches in Melbourne, one in Queensland 
--                                                            and the other two are in South AU and Western AU.
--Once there are more than one branch in one area, there would be a head branch which is in charge of the other branches in that area.
--The head branch would have departments such as President office, Information Technology, Human resource and Creditcard department.
--And the other branches would have Administration Office instead. All the branches would have personal banking and corporate banking department.
--Bank of China has built a fast and highly efficient global service network supplying a diversified range of financial products.
--Financial services include:Deposit products denominated in AUD and foreign currency;
--                           Corporate financing;
--                           ATM and internet banking;
--                           Bank cards;
--                           Personal financing.
--One branch should have one to many ATMs. In this databse, those ATM's brands are assumed to be Pacific or Sun. 
--A customer could open a bankcard at any branch and then use this bankcard to make transactions at any branch as they want.
--Customer can also have a transcation using cash which means a bankcard is not a must.
--Only staff from personal banking department could provide peronal banking services for customers.
--And only Corporate banking staff could serve company customers.
--

DROP table BOC_AU CASCADE;
DROP table BOC_ATM CASCADE;
DROP table BOC_Department CASCADE;
DROP table BOC_Staff CASCADE;
DROP table BOC_Customer CASCADE;
DROP table BOC_BankCard CASCADE;
DROP table BOC_Transaction CASCADE;

Create table BOC_AU
(
        BranchID       integer    NOT NULL,   
        BranchName     varchar    NOT NULL,
        TEL            integer    NOT NULL,
        FAX            integer    NOT NULL,
        Email          TEXT       NOT NULL,
        Address        TEXT       NOT NULL,
        HeadBranchID   integer,
    CONSTRAINT BOC_AUPK PRIMARY KEY(BranchID),
    CONSTRAINT BOC_AUFK FOREIGN KEY(HeadBranchID) REFERENCES BOC_AU

);
Create table BOC_ATM
(
        ATMID        integer    NOT NULL,
        ATMBrand     varchar    NOT NULL,
        BranchID     integer    NOT NULL,
    CONSTRAINT BOC_ATMPK PRIMARY KEY(ATMID),
    CONSTRAINT BOC_ATMFK FOREIGN KEY(BranchID) REFERENCES BOC_AU ON DELETE RESTRICT,
    CONSTRAINT di_table_BOC_ATM_ATMBrand CHECK (ATMBrand IN ('Pacific','Sun'))

);
Create table BOC_Department
(
        BranchID     integer    NOT NULL,
        DepID        integer    NOT NULL,
        DepName      TEXT       NOT NULL,
    CONSTRAINT BOC_DepartmentPK PRIMARY KEY(BranchID,DepID),
    CONSTRAINT BOC_DepartmentFK FOREIGN KEY(BranchID) REFERENCES BOC_AU ON DELETE RESTRICT
 
);
Create table BOC_Staff
(
        BranchID    integer,
        DepID       integer,
        StaffID     integer,
        StaffName   TEXT,
        StaffGender char(1),
        StaffBirth  integer,
        StaffLevel  integer,
        Salary      integer,
        Address     TEXT,
        ManagerID   integer,
    CONSTRAINT BOC_StaffPK PRIMARY KEY(StaffID),
    CONSTRAINT BOC_StaffFK FOREIGN KEY(BranchID,DepID) REFERENCES BOC_Department ON DELETE RESTRICT,
    CONSTRAINT di_table_BOC_Staff_StaffGender CHECK(StaffGender IN ('M','F')),
    CONSTRAINT di_table_BOC_Staff_StaffBirth CHECK(StaffBirth >= 1900 AND StaffBirth <= 2000),
    CONSTRAINT di_table_BOC_Staff_StaffLevel CHECK(StaffLevel >=1 and StaffLevel < 10)
 
);
Create table BOC_Customer
(
        CusID      integer,    
        CusName    TEXT,
        CusGender  char(1),   
        CusBirth   integer,
        CusTEL     integer,
        CusType    char(1),
        CusLevel   char(1),
        Address    TEXT,
    CONSTRAINT BOC_CustomerPK PRIMARY KEY(CusID),
    CONSTRAINT di_table_BOC_Customer_CusGender CHECK (CusGender IN ('M',   --male
                                                                    'F')), --female
    CONSTRAINT di_table_BOC_Customer_CusType CHECK (CusType IN ('P',    --private,
                                                                'C')),  --company
    CONSTRAINT di_table_BOC_Customer_CusLevel CHECK (CusLevel IN ('A','B','C','N')),
    CONSTRAINT di_table_BOC_Customer_CusBirth CHECK (CusBirth >= 1900 AND CusBirth <= 2000)
);
Create table BOC_BankCard
(
        BranchID    integer,
        CusID       integer,
        CardNo      integer,
    CONSTRAINT BOC_BankCardPK PRIMARY KEY(BranchID,CusID,CardNO),
    CONSTRAINT BOC_BankCardFK_AU FOREIGN KEY(BranchID) REFERENCES BOC_AU ON DELETE RESTRICT,
    CONSTRAINT BOC_BankCardFK_Customer FOREIGN KEY(CusID) REFERENCES BOC_Customer ON DELETE CASCADE
);

Create table BOC_Transaction
(
        TransID    integer,
        CardNO     integer,
        CusID      integer,
        StaffID    integer,
        TransType  TEXT,
        Amount     integer,
        Year       integer,
        Month      char(3),
        Day        integer,
    CONSTRAINT BOC_TransactionPK PRIMARY KEY(TransID),
    CONSTRAINT BOC_TransactionFK_Customer FOREIGN KEY(CusID) REFERENCES BOC_Customer ON DELETE CASCADE,
    CONSTRAINT BOC_TransactionFK_Staff FOREIGN KEY(StaffID) REFERENCES BOC_Staff ON DELETE RESTRICT,
    CONSTRAINT di_table_BOC_Transaction_TransType CHECK 
                       (TransType IN ('O',  --open account
                                      'W',  --withdraw money
                                      'D',  --deposit money
                                      'E',  --exchange money
                                      'C')),--company transcations
    CONSTRAINT di_table_BOC_Transaction_Year CHECK (Year >= 1990 AND Year <= 2018),
    CONSTRAINT di_table_BOC_Transaction_Month CHECK
                       (Month IN('Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec')),
    CONSTRAINT di_table_BOC_Transaction_Day CHECK (Day >= 1 AND Day <= 31)
);


CREATE VIEW PersonalBankingStaff As
            select staffid as ID, staffname as name,staffgender as G,(2018 -staffbirth) as Age,stafflevel as level,salary,depid 
            from boc_staff natural join boc_department 
            where depname = 'Personal Banking' order by staffid;


CREATE VIEW SydneyStaff As
             select s.staffid as id, s.staffname as name,s.stafflevel,(2018-s.staffbirth) as age,d.depname,a.branchid,a.branchname 
             from boc_au a, boc_staff s,boc_department d  where( a.branchid = 1 or a.headbranchid = 1) 
                                and a.branchid = s.branchid and a.branchid = d.branchid and s.branchid = d.branchid 
                                and d.depid = s.depid  order by a.branchname;
